An object is constructed but not used

This happens if an object is constructed using the new operator and its reference is not assigned to a variable. 


var persons = 0;

function Person(name) {

    this.name = name;
    persons += 1;
}

var p1 = new Person("Sokratis");
new Person();

In terms of syntax this is not incorrect but it is certainly not an efficient way to use a constructor. If the only thing needed from the construction is its side effects, then the related part should be removed from the constructor: 


var persons = 0;

function Person(name) {

    this.name = name;
}

var p1 = new Person("Sokratis");
persons += 1;
